def remove_chars(input_string, chars_to_remove): result = ''.join(char for char in input_string if char not in chars_to_remove) return result #використання input_string = input("Введіть рядок: ") chars_to_remove = list(input("Введіть символи для видалення (без пробілів): ")) # Викликаємо функцію та результат print(remove_chars(input_string, chars_to_remove))
